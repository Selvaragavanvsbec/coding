class Solution {
    public boolean hasDuplicate(int[] nums) {
        if(nums.length<=1){
            return false;
        }
        HashSet<Integer> set=new HashSet<>();
        set.add(nums[0]);
        for(int i=1;i<nums.length;i++){
            if(set.contains(nums[i])){
                return true;
                
            }
            else{
                set.add(nums[i]);
               
            }
        }
    return false;
    }
}
